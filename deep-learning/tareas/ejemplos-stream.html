Reconocimiento de Imágenes
import cv2
import tensorflow as tf

# Cargar modelo preentrenado
model = tf.keras.applications.MobileNetV2(
    weights='imagenet',
    include_top=True
)

# Procesar imagen
def analizar_imagen(ruta_imagen):
    img = cv2.imread(ruta_imagen)
    img = cv2.resize(img, (224, 224))
    img = tf.expand_dims(img, 0)
    
    prediccion = model.predict(img)
    return tf.keras.applications.mobilenet_v2.decode_predictions(
        prediccion, top=3
    )[0]

# Ejemplo: foto de empanadas
resultado = analizar_imagen("empanadas.jpg")
# Output: [('empanada', 0.89), ('pastry', 0.07), ...]
Procesamiento de Texto Chileno
Diccionario de Chilenismos
chilenismos = {
    'bacán': 'genial, excelente',
    'fome': 'aburrido',
    'pololo/a': 'novio/a',
    'carrete': 'fiesta',
    'cachai': '¿entiendes?',
    'al tiro': 'inmediatamente',
    'cuático': 'increíble',
    'choro': 'genial, bueno'
}

def detectar_chilenismos(texto):
    palabras_encontradas = []
    for palabra, significado in chilenismos.items():
        if palabra in texto.lower():
            palabras_encontradas.append({
                'palabra': palabra,
                'significado': significado
            })
    return palabras_encontradas
Análisis de Frecuencia
import matplotlib.pyplot as plt
import seaborn as sns
from collections import Counter

def analizar_frecuencia_palabras(textos):
    todas_palabras = []
    for texto in textos:
        palabras = texto.lower().split()
        todas_palabras.extend(palabras)
    
    frecuencia = Counter(todas_palabras)
    
    # Visualizar
    plt.figure(figsize=(10, 6))
    palabras_comunes = frecuencia.most_common(10)
    
    sns.barplot(
        x=[p[1] for p in palabras_comunes],
        y=[p[0] for p in palabras_comunes]
    )
    plt.title('Palabras más frecuentes en chats chilenos')
    plt.show()
    
    return frecuencia

¿Por qué no son intercambiables?

Procesamiento de Texto
RNN es Superior

¿Por qué?

  • El texto tiene orden secuencial
  • Las palabras dependen del contexto anterior
  • La memoria es crucial para entender significado
Ejemplo Chileno:
"El partido estuvo bacán, pero al final quedé fome"
RNN entiende que "bacán" se refiere al partido y "fome" al estado final
CNN Limitada

CNN ve palabras como píxeles independientes, perdiendo el contexto secuencial

Procesamiento de Video
Híbrido: CNN + RNN

Mejor Enfoque:

  • CNN: Extrae características de cada frame
  • RNN: Conecta la secuencia temporal

Frame 1 CNN → Jugador con balón
Frame 2 CNN → Movimiento hacia arco
Frame 3 CNN → Gol celebrado
RNN → Secuencia completa: "Jugada de gol"
Solo CNN

Vería frames independientes, sin entender la secuencia de la jugada

Razones:

  • Entonación y ritmo son secuenciales

CNN para Análisis de Imágenes

Caso: Detectar comida chilena en fotos

import tensorflow as tf
from tensorflow.keras import layers

# CNN para clasificar comida chilena
model = tf.keras.Sequential([
    # Extracción de características espaciales
    layers.Conv2D(32, (3,3), activation='relu', 
                  input_shape=(224, 224, 3)),
    layers.MaxPooling2D(2, 2),
    
    layers.Conv2D(64, (3,3), activation='relu'),
    layers.MaxPooling2D(2, 2),
    
    layers.Conv2D(128, (3,3), activation='relu'),
    layers.MaxPooling2D(2, 2),
    
    # Clasificación
    layers.Flatten(),
    layers.Dense(512, activation='relu'),
    layers.Dense(5, activation='softmax')  # 5 comidas chilenas
])

# Clases: empanada, completo, cazuela, pastel_choclo, sopaipilla
clases = ['empanada', 'completo', 'cazuela', 
          'pastel_choclo', 'sopaipilla']

# ¿Por qué CNN?
# ✅ Detecta bordes, texturas, formas
# ✅ Invariante a posición en la imagen
# ✅ Jerarquía: bordes → texturas → objetos
¿Por qué no RNN aquí?
Una imagen no tiene secuencia temporal. Los píxeles no dependen del "orden" como las palabras en una oración.
RNN para Análisis de Texto

Caso: Analizar sentimientos en comentarios chilenos

import tensorflow as tf
from tensorflow.keras import layers

# RNN para análisis de sentimientos
model = tf.keras.Sequential([
    # Embedding de palabras
    layers.Embedding(vocab_size, 128),
    
    # Procesamiento secuencial
    layers.LSTM(64, return_sequences=True),
    layers.Dropout(0.5),
    
    layers.LSTM(32),
    layers.Dropout(0.5),
    
    # Clasificación de sentimiento
    layers.Dense(3, activation='softmax')  # positivo, neutro, negativo
])

# Ejemplo de texto chileno:
texto = "El partido estuvo bacán, pero el árbitro estuvo fome"

# Procesamiento secuencial:
# "El" → contexto inicial
# "partido" → tema deportivo
# "estuvo" → verbo de estado
# "bacán" → sentimiento positivo (chilenismo)
# "pero" → conector de contraste
# "árbitro" → nuevo sujeto
# "fome" → sentimiento negativo (chilenismo)

# ¿Por qué RNN?
# ✅ Entiende que "bacán" se refiere al partido
# ✅ "pero" indica cambio de sentimiento
# ✅ "fome" se refiere al árbitro, no al partido
¿Por qué no CNN aquí?
CNN vería "bacán" y "fome" como características independientes, sin entender que se refieren a sujetos diferentes.
Enfoque Híbrido: CNN + RNN para Video

Caso: Analizar jugadas de fútbol chileno en video

import tensorflow as tf
from tensorflow.keras import layers

# Arquitectura híbrida para análisis de video
def create_video_model():
    # Entrada: secuencia de frames (tiempo, alto, ancho, canales)
    video_input = layers.Input(shape=(None, 224, 224, 3))
    
    # CNN para cada frame (extracción de características espaciales)
    cnn_base = tf.keras.applications.ResNet50(
        weights='imagenet', 
        include_top=False,
        input_shape=(224, 224, 3)
    )
    
    # Aplicar CNN a cada frame usando TimeDistributed
    cnn_features = layers.TimeDistributed(cnn_base)(video_input)
    cnn_features = layers.TimeDistributed(layers.GlobalAveragePooling2D())(cnn_features)
    
    # RNN para secuencia temporal
    lstm_out = layers.LSTM(128, return_sequences=True)(cnn_features)
    lstm_out = layers.LSTM(64)(lstm_out)
    
    # Clasificación de jugada
    output = layers.Dense(4, activation='softmax')(lstm_out)
    
    model = tf.keras.Model(video_input, output)
    return model

# Clases de jugadas
jugadas = ['gol', 'falta', 'corner', 'fuera_juego']

# ¿Por qué este enfoque híbrido?
# CNN: Extrae características de cada frame
#      - Detecta jugadores, balón, arco
#      - Reconoce posiciones y movimientos
# 
# RNN: Conecta la secuencia temporal
#      - Entiende la progresión de la jugada
#      - Relaciona frames consecutivos
#      - Predice el resultado de la secuencia

Limitaciones y Casos de Uso

Cuándo usar cada arquitectura y sus limitaciones

Limitaciones de CNN
❌ No maneja secuencias temporales

Ejemplo: "No me gusta" vs "No, me gusta" - CNN no distingue el orden

❌ Sin memoria entre entradas

Cada imagen se procesa independientemente

❌ Requiere tamaño fijo de entrada

Textos de diferente longitud necesitan padding

✅ Usar CNN cuando:
  • Datos tienen estructura espacial (imágenes)
  • Patrones son locales e invariantes
  • No hay dependencias temporales
  • Necesitas paralelización eficiente
Ejemplos ideales para CNN:
  • 🖼️ Clasificación de imágenes de comida chilena
  • 🏥 Diagnóstico médico por rayos X
  • 🛰️ Análisis de imágenes satelitales de Chile
  • 👁️ Reconocimiento facial
Limitaciones de RNN
❌ Problema del gradiente que desaparece

Dificultad para aprender dependencias largas

❌ Entrenamiento secuencial lento

No se puede paralelizar como CNN

❌ Memoria limitada

Olvida información de secuencias muy largas

✅ Usar RNN cuando:
  • Datos son secuenciales (texto, audio, tiempo)
  • El orden importa para el significado
  • Necesitas memoria del contexto anterior
  • Longitud de entrada es variable
Ejemplos ideales para RNN:
  • 💬 Análisis de sentimientos en redes sociales
  • 🎵 Transcripción de audio a texto
  • 📈 Predicción de series temporales (bolsa)
  • 🤖 Chatbots conversacionales

Redes Neuronales Recurrentes (RNN)

Análisis matemático y aplicaciones prácticas en el contexto chileno

Fundamentos Matemáticos de RNN

📐 Ecuaciones Fundamentales
Estado Oculto en tiempo t:
ht = tanh(Whh · ht-1 + Wxh · xt + bh)
Donde:
• ht = Estado oculto actual
• ht-1 = Estado oculto anterior
• xt = Entrada actual
• Whh = Pesos recurrentes
• Wxh = Pesos de entrada
• bh = Sesgo
Salida en tiempo t:
yt = Why · ht + by
Donde:
• yt = Salida en tiempo t
• Why = Pesos de salida
• by = Sesgo de salida
Interpretación Didáctica

La RNN "recuerda" información previa (ht-1) y la combina con la entrada actual (xt) para generar un nuevo estado de memoria (ht).

Variantes de RNN y sus Fórmulas

LSTM (Long Short-Term Memory)
Compuertas LSTM:
1. Compuerta de Olvido:
ft = σ(Wf · [ht-1, xt] + bf)
2. Compuerta de Entrada:
it = σ(Wi · [ht-1, xt] + bi)
3. Estado Candidato:
C̃t = tanh(WC · [ht-1, xt] + bC)
4. Estado de Celda:
Ct = ft * Ct-1 + it * C̃t
5. Compuerta de Salida:
ot = σ(Wo · [ht-1, xt] + bo)
ht = ot * tanh(Ct)
🎯 Ventaja Clave:

Resuelve el problema del gradiente que desaparece mediante las compuertas que controlan el flujo de información.

GRU (Gated Recurrent Unit)
Compuertas GRU (Simplificadas):
1. Compuerta de Reset:
rt = σ(Wr · [ht-1, xt])
2. Compuerta de Actualización:
zt = σ(Wz · [ht-1, xt])
3. Estado Candidato:
h̃t = tanh(W · [rt * ht-1, xt])
4. Estado Final:
ht = (1 - zt) * ht-1 + zt * h̃t
⚡ Ventaja Clave:

Menos parámetros que LSTM pero rendimiento similar. Más eficiente computacionalmente.

import tensorflow as tf
from tensorflow.keras import layers, models
import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split
import re

# ===== PREPROCESAMIENTO ESPECÍFICO PARA ESPAÑOL CHILENO =====
class ChileanTextPreprocessor:
    def __init__(self):
        # Diccionario de chilenismos y su polaridad
        self.chilenismos = {
            'bacán': 'positivo', 'genial': 'positivo', 'la raja': 'positivo',
            'fome': 'negativo', 'flaite': 'negativo', 'como las weas': 'negativo',
            'cuático': 'neutro', 'brigido': 'intenso', 'terrible': 'intenso'
        }
        
        # Normalizaciones comunes
        self.normalizations = {
            'weon': 'persona', 'weón': 'persona', 'wn': 'persona',
            'po': '', 'poh': '', 'cachai': 'entiendes',
            'altiro': 'inmediatamente', 'al tiro': 'inmediatamente'
        }
    
    def preprocess_text(self, text):
        # Convertir a minúsculas
        text = text.lower()
        
        # Normalizar chilenismos
        for chileno, normal in self.normalizations.items():
            text = text.replace(chileno, normal)
        
        # Limpiar caracteres especiales pero mantener emojis
        text = re.sub(r'[^\w\s😀-🙏]', '', text)
        
        return text

# ===== ARQUITECTURA RNN ESPECÍFICA =====
def create_chilean_sentiment_model(vocab_size, embedding_dim=128, max_length=100):
    """
    Modelo RNN optimizado para análisis de sentimientos en español chileno
    """
    model = models.Sequential([
        # Capa de embedding para palabras
        layers.Embedding(
            input_dim=vocab_size, 
            output_dim=embedding_dim,
            input_length=max_length,
            mask_zero=True  # Ignora padding
        ),
        
        # Dropout para regularización
        layers.Dropout(0.3),
        
        # LSTM Bidireccional para capturar contexto en ambas direcciones
        layers.Bidirectional(
            layers.LSTM(64, return_sequences=True, dropout=0.3, recurrent_dropout=0.3)
        ),
        
        # Segunda capa LSTM para patrones más complejos
        layers.Bidirectional(
            layers.LSTM(32, dropout=0.3, recurrent_dropout=0.3)
        ),
        
        # Capas densas para clasificación
        layers.Dense(64, activation='relu'),
        layers.Dropout(0.5),
        layers.Dense(32, activation='relu'),
        layers.Dropout(0.3),
        
        # Salida: 3 clases (positivo, neutro, negativo)
        layers.Dense(3, activation='softmax')
    ])
    
    return model

# ===== FUNCIÓN DE PÉRDIDA PERSONALIZADA =====
def weighted_categorical_crossentropy(class_weights):
    """
    Función de pérdida que penaliza más los errores en clases desbalanceadas
    """
    def loss_function(y_true, y_pred):
        # Convertir a tensores
        y_true = tf.cast(y_true, tf.float32)
        y_pred = tf.cast(y_pred, tf.float32)
        
        # Aplicar pesos por clase
        weights = tf.reduce_sum(class_weights * y_true, axis=1)
        
        # Calcular cross-entropy
        cross_entropy = -tf.reduce_sum(y_true * tf.math.log(y_pred + 1e-8), axis=1)
        
        # Aplicar pesos
        weighted_loss = cross_entropy * weights
        
        return tf.reduce_mean(weighted_loss)
    
    return loss_function

# ===== MÉTRICAS PERSONALIZADAS =====
class ChileanSentimentMetrics:
    @staticmethod
    def sentiment_accuracy(y_true, y_pred):
        """Precisión específica para sentimientos"""
        return tf.keras.metrics.categorical_accuracy(y_true, y_pred)
    
    @staticmethod
    def polaridad_f1_score(y_true, y_pred):
        """F1-Score para polaridad (positivo vs negativo, ignorando neutro)"""
        # Convertir a binario: positivo=1, negativo/neutro=0
        y_true_binary = tf.cast(tf.argmax(y_true, axis=1) == 0, tf.float32)
        y_pred_binary = tf.cast(tf.argmax(y_pred, axis=1) == 0, tf.float32)
        
        # Calcular F1
        precision = tf.keras.metrics.Precision()
        recall = tf.keras.metrics.Recall()
        
        precision.update_state(y_true_binary, y_pred_binary)
        recall.update_state(y_true_binary, y_pred_binary)
        
        p = precision.result()
        r = recall.result()
        
        return 2 * (p * r) / (p + r + 1e-8)

# ===== ENTRENAMIENTO CON DATOS CHILENOS =====
def train_chilean_sentiment_model():
    # Datos de ejemplo (en la práctica, cargar desde archivo)
    textos_ejemplo = [
        "El completo italiano está bacán, me encanta",  # Positivo
        "El metro anda como las weas otra vez",         # Negativo
        "Hoy es un día normal, nada especial",          # Neutro
        "Qué fome la película, terrible mala",          # Negativo
        "La raja el concierto, estuvo cuático",         # Positivo
    ]
    
    sentimientos = [
        [1, 0, 0],  # Positivo
        [0, 0, 1],  # Negativo
        [0, 1, 0],  # Neutro
        [0, 0, 1],  # Negativo
        [1, 0, 0],  # Positivo
    ]
    
    # Preprocesamiento
    preprocessor = ChileanTextPreprocessor()
    textos_procesados = [preprocessor.preprocess_text(texto) for texto in textos_ejemplo]
    
    # Tokenización (simplificada para el ejemplo)
    tokenizer = tf.keras.preprocessing.text.Tokenizer(num_words=5000)
    tokenizer.fit_on_texts(textos_procesados)
    
    # Convertir a secuencias
    sequences = tokenizer.texts_to_sequences(textos_procesados)
    data = tf.keras.preprocessing.sequence.pad_sequences(sequences, maxlen=100)
    
    # Crear modelo
    vocab_size = len(tokenizer.word_index) + 1
    model = create_chilean_sentiment_model(vocab_size)
    
    # Pesos para clases desbalanceadas (común en datos reales)
    class_weights = tf.constant([1.2, 1.0, 1.3])  # [pos, neu, neg]
    
    # Compilar modelo
    model.compile(
        optimizer=tf.keras.optimizers.Adam(learning_rate=0.001),
        loss=weighted_categorical_crossentropy(class_weights),
        metrics=[
            'accuracy',
            ChileanSentimentMetrics.sentiment_accuracy,
            ChileanSentimentMetrics.polaridad_f1_score
        ]
    )
    
    # Callbacks para entrenamiento
    callbacks = [
        tf.keras.callbacks.EarlyStopping(
            monitor='val_loss', 
            patience=5, 
            restore_best_weights=True
        ),
        tf.keras.callbacks.ReduceLROnPlateau(
            monitor='val_loss', 
            factor=0.5, 
            patience=3
        )
    ]
    
    return model, tokenizer, preprocessor

# ===== FUNCIÓN DE PREDICCIÓN =====
def predecir_sentimiento_chileno(texto, model, tokenizer, preprocessor):
    """
    Predice el sentimiento de un texto en español chileno
    """
    # Preprocesar
    texto_procesado = preprocessor.preprocess_text(texto)
    
    # Tokenizar
    sequence = tokenizer.texts_to_sequences([texto_procesado])
    padded = tf.keras.preprocessing.sequence.pad_sequences(sequence, maxlen=100)
    
    # Predecir
    prediccion = model.predict(padded)[0]
    
    # Interpretar resultado
    clases = ['Positivo', 'Neutro', 'Negativo']
    sentimiento = clases[np.argmax(prediccion)]
    confianza = np.max(prediccion)
    
    return {
        'texto_original': texto,
        'texto_procesado': texto_procesado,
        'sentimiento': sentimiento,
        'confianza': f"{confianza:.2%}",
        'probabilidades': {
            'Positivo': f"{prediccion[0]:.2%}",
            'Neutro': f"{prediccion[1]:.2%}",
            'Negativo': f"{prediccion[2]:.2%}"
        }
    }

Fórmulas de Referencia Rápida
RNN Básica
ht = tanh(Whhht-1 + Wxhxt + b)
yt = Whyht + by
LSTM Gates
Forget: ft = σ(Wf[ht-1,xt] + bf)
Input: it = σ(Wi[ht-1,xt] + bi)
Output: ot = σ(Wo[ht-1,xt] + bo)
Cell: Ct = ft*Ct-1 + it*C̃t
GRU Gates
Reset: rt = σ(Wr[ht-1,xt])
Update: zt = σ(Wz[ht-1,xt])
New: h̃t = tanh(W[rt*ht-1,xt])
Final: ht = (1-zt)*ht-1 + zt*h̃t
Funciones Activación
Sigmoid: σ(x) = 1/(1+e-x)
Tanh: tanh(x) = (ex-e-x)/(ex+e-x)
ReLU: ReLU(x) = max(0,x)
Softmax: S(xi) = exi/Σexj

Resumen: ¿Qué Aprendimos?
Procesamiento Secuencial

La RNN procesa palabra por palabra, manteniendo memoria del contexto anterior en cada paso.

Contexto Acumulativo

Cada palabra contribuye al estado oculto, creando una representación rica del significado completo.

Chilenismos Clave

Palabras como "bacán" y "fome" tienen embeddings especializados que capturan su significado cultural.

El último estado oculto contiene toda la información necesaria para la clasificación final.
