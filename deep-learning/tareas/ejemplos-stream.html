import tensorflow as tf
from tensorflow.keras import layers

# Arquitectura h√≠brida para an√°lisis de video
def create_video_model():
    # Entrada: secuencia de frames (tiempo, alto, ancho, canales)
    video_input = layers.Input(shape=(None, 224, 224, 3))
    
    # CNN para cada frame (extracci√≥n de caracter√≠sticas espaciales)
    cnn_base = tf.keras.applications.ResNet50(
        weights='imagenet', 
        include_top=False,
        input_shape=(224, 224, 3)
    )
    
    # Aplicar CNN a cada frame usando TimeDistributed
    cnn_features = layers.TimeDistributed(cnn_base)(video_input)
    cnn_features = layers.TimeDistributed(layers.GlobalAveragePooling2D())(cnn_features)
    
    # RNN para secuencia temporal
    lstm_out = layers.LSTM(128, return_sequences=True)(cnn_features)
    lstm_out = layers.LSTM(64)(lstm_out)
    
    # Clasificaci√≥n de jugada
    output = layers.Dense(4, activation='softmax')(lstm_out)
    
    model = tf.keras.Model(video_input, output)
    return model

# Clases de jugadas

# ===== PREPROCESAMIENTO ESPEC√çFICO PARA ESPA√ëOL CHILENO =====
class ChileanTextPreprocessor:
    def __init__(self):
        # Diccionario de chilenismos y su polaridad
        self.chilenismos = {
            'bac√°n': 'positivo', 'genial': 'positivo', 'la raja': 'positivo',
            'fome': 'negativo', 'flaite': 'negativo', 'como las weas': 'negativo',
            'cu√°tico': 'neutro', 'brigido': 'intenso', 'terrible': 'intenso'
        }
        
        # Normalizaciones comunes
        self.normalizations = {
            'weon': 'persona', 'we√≥n': 'persona', 'wn': 'persona',
            'po': '', 'poh': '', 'cachai': 'entiendes',
            'altiro': 'inmediatamente', 'al tiro': 'inmediatamente'
        }
    
    def preprocess_text(self, text):
        # Convertir a min√∫sculas
        text = text.lower()
        
        # Normalizar chilenismos
        for chileno, normal in self.normalizations.items():
            text = text.replace(chileno, normal)
        
        # Limpiar caracteres especiales pero mantener emojis
        text = re.sub(r'[^\w\süòÄ-üôè]', '', text)
        
        return text

# ===== ARQUITECTURA RNN ESPEC√çFICA =====
def create_chilean_sentiment_model(vocab_size, embedding_dim=128, max_length=100):
    """
    Modelo RNN optimizado para an√°lisis de sentimientos en espa√±ol chileno
    """
    model = models.Sequential([
        # Capa de embedding para palabras
        layers.Embedding(
            input_dim=vocab_size, 
            output_dim=embedding_dim,
            input_length=max_length,
            mask_zero=True  # Ignora padding
        ),
        
        # Dropout para regularizaci√≥n
        layers.Dropout(0.3),
        
        # LSTM Bidireccional para capturar contexto en ambas direcciones
        layers.Bidirectional(
            layers.LSTM(64, return_sequences=True, dropout=0.3, recurrent_dropout=0.3)
        ),
        
        # Segunda capa LSTM para patrones m√°s complejos
        layers.Bidirectional(
            layers.LSTM(32, dropout=0.3, recurrent_dropout=0.3)
        ),
        
        # Capas densas para clasificaci√≥n
        layers.Dense(64, activation='relu'),
        layers.Dropout(0.5),
        layers.Dense(32, activation='relu'),
        layers.Dropout(0.3),
        
        # Salida: 3 clases (positivo, neutro, negativo)
        layers.Dense(3, activation='softmax')
    ])
    
    return model

# ===== FUNCI√ìN DE P√âRDIDA PERSONALIZADA =====
def weighted_categorical_crossentropy(class_weights):
    """
    Funci√≥n de p√©rdida que penaliza m√°s los errores en clases desbalanceadas
    """
    def loss_function(y_true, y_pred):
        # Convertir a tensores
        y_true = tf.cast(y_true, tf.float32)
        y_pred = tf.cast(y_pred, tf.float32)
        
        # Aplicar pesos por clase
        weights = tf.reduce_sum(class_weights * y_true, axis=1)
        
        # Calcular cross-entropy
        cross_entropy = -tf.reduce_sum(y_true * tf.math.log(y_pred + 1e-8), axis=1)
        
        # Aplicar pesos
        weighted_loss = cross_entropy * weights
        
        return tf.reduce_mean(weighted_loss)
    
    return loss_function

# ===== M√âTRICAS PERSONALIZADAS =====
class ChileanSentimentMetrics:
    @staticmethod
    def sentiment_accuracy(y_true, y_pred):
        """Precisi√≥n espec√≠fica para sentimientos"""
        return tf.keras.metrics.categorical_accuracy(y_true, y_pred)
    
    @staticmethod
    def polaridad_f1_score(y_true, y_pred):
        """F1-Score para polaridad (positivo vs negativo, ignorando neutro)"""
        # Convertir a binario: positivo=1, negativo/neutro=0
        y_true_binary = tf.cast(tf.argmax(y_true, axis=1) == 0, tf.float32)
        y_pred_binary = tf.cast(tf.argmax(y_pred, axis=1) == 0, tf.float32)
        
        # Calcular F1
        precision = tf.keras.metrics.Precision()
        recall = tf.keras.metrics.Recall()
        
        precision.update_state(y_true_binary, y_pred_binary)
        recall.update_state(y_true_binary, y_pred_binary)
        
        p = precision.result()
        r = recall.result()
        
        return 2 * (p * r) / (p + r + 1e-8)

def predecir_sentimiento_chileno(texto, model, tokenizer, preprocessor):
    """
    Predice el sentimiento de un texto en espa√±ol chileno
    """
    # Preprocesar
    texto_procesado = preprocessor.preprocess_text(texto)
    
    # Tokenizar
    sequence = tokenizer.texts_to_sequences([texto_procesado])
    padded = tf.keras.preprocessing.sequence.pad_sequences(sequence, maxlen=100)
    
    # Predecir
    prediccion = model.predict(padded)[0]
    
    # Interpretar resultado
    clases = ['Positivo', 'Neutro', 'Negativo']
    sentimiento = clases[np.argmax(prediccion)]
    confianza = np.max(prediccion)
    
    return {
        'texto_original': texto,
        'texto_procesado': texto_procesado,
        'sentimiento': sentimiento,
        'confianza': f"{confianza:.2%}",
        'probabilidades': {
            'Positivo': f"{prediccion[0]:.2%}",
            'Neutro': f"{prediccion[1]:.2%}",
            'Negativo': f"{prediccion[2]:.2%}"
        }
    }
